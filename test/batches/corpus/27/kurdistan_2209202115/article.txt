د. دیندار زێباری، رێکخەری راسپاردە نێودەوڵەتییەکان لە حکومەتی هەرێمی کوردستان بە تۆڕی میدیایی رووداوی راگەیاند، لەو کاتەوەی کە شەڕ و پێکدان و بۆردوومانەکان لە سنوورەکانی هەرێمی کوردستان دەستیان پێکردووە، بووەتە هۆی ئاوارەبوونی سەتان هاونیشتمانی و زیان گەیاندن بە گیان و موڵک و ماڵیان. هاوکات، هیچ لایەنێکی نیشتمانی و نێودەوڵەتی دەستپێشخەر نەبوونە بۆ گەیشتن بە دەرئەنجامێک بۆ قەرەبووکردنەوەی ئەو هاونیشتمانییانەی کە موڵک و ماڵیان وێران بووە، زەوی و زاریان زیانی پێگەیشتووە، و ژیانیان لە مەترسیدا بووە.  د. دیندار زێباری ئاماژەی بەوەکرد، "هێرش و بەرەنگارییەکان بوونەتە هۆی ئەوەی کە دانیشتووانی ئەم ناوچانە بە کۆمەڵ ئاوارەبن، لە نێوان مانگی کانوونی دووەم بۆ ئابی ساڵی ٢٠٢١ دا، ١٦٥ هێرشی ئاسمانی ئەنجامدراون، ٢٧٤ گولەتۆپ تەقێندراون و شەش هێرشی زەوینیش ئەنجامدراون. بۆردوومانەکان ناوچەکانی پێنجوێن، چوارتا، هەوارە کۆنی سەربە پارێزگای سلێمانی، شنگال، ئەسکبلان، شیلانی، کلکا نسارا، کلکا ئیسلام، کشانی، هرووری، کورزار، دشیش، سکیر، ئاورا، باپیری، هروور، خامتیر، کەلها شابانی، لیفانی، دیری، هفشن، بیدهوهی، بونکستای سەر بە پارێزگای دهۆکی کردوونەتە ئامانج. "  رێکخەری راسپاردە نێودەوڵەتییەکان لە حکومەتی هەرێمی کوردستان ئاماژەی بەوەشکرد، لە رێگەی هێرشی ئاسمانی و تۆپباران گوندەکانی سنووری ناحیەی سیدەکانی سەر بە پارێزگای هەولێر بۆردوومانکراون، لەگەڵ ئەوەش، تۆپخانەکان گوندەکانی نزیک ناوچەکانی باتووفە، کانی ماسی، درکاری سەر بە پارێزگای دهۆکی لە سەرەتای مانگی ئەیلوولی ٢٠٢١ دا تۆپباران کرد.  بەپێی قسەی رێکخەری راسپاردە نێودەوڵەتییەکان لە حکومەتی هەرێمی کوردستان، "ئەم هێرشانە مەترسی دەخەنە سەر گیانی هاونیشتمانییانی هەرێمی کوردستان، زیان دەگەیێنن بە موڵک و ماڵ و زەوی و زاری جووتیاران، بینا و خانووەکان وێران دەکەن و زیان دەگەیێنن بە دارستانەکان. سەرەڕای ئەمەش، چەندین دانیشتووی گووند نشینەکان ئاوارە بوونە بەهۆی ئەم بۆردوومانانە و چەندین هاونیشتمانی گیانیان لەدەستداوە و دەیانی دیکە بوونەتە قوربانی. سەرباری ئەمەش، بەهۆی شەڕ و پێکدادانەکان نێوان پەکەکە و سوپای تورکیا لە ماوەی سێ دەیەی رابردوودا زیاتر لە ٨٠٠ گوند چۆڵکراون و بێ خزمەتگوزاری ماونەتەوە".  دیندار زێباری داوای کۆتایی هێنان بە گرژی و ئاڵۆزییەکان لە ناوچە سنورییەکان دەکات و دەشڵێت، "داوا لە وڵاتانی دراوسێ دەکەین کە رەچاوی بارودۆخی هاونیشتمانییانی لە ناوچەکانی سەر سنوور بکەن کە ئێستا رووبەڕووی بۆردوومانەکان بوونەتەوە و ناچارن ماڵەکانیان جێبهێڵن و دەست لە موڵک و ماڵەکانیان هەڵبگرن. "
