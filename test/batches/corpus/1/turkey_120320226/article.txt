له رێوڕه‌سمى كردنه‌وه‌ى كۆڕبه‌نده‌كه‌دا هه‌ريه‌ك له ره‌جه‌ب ته‌ييب ئه‌ردۆغان، سه‌رۆکكۆمارى توركيا؛ مه‌ولوود چاوشئۆغڵو، وه‌زيرى ده‌ره‌وه‌ى توركيا و ئه‌نتۆنيۆ گۆتێرێس، سكرتێرى گشتيى نه‌ته‌وه يه‌كگرتووه‌كان گوتاريان پێشكێش كرد و تيشك خرايه سه‌ر ئاڵنگاری و كێشه‌كانى جيهان، شه‌ڕ و پێكدادانى رووسيا و ئۆكراینا، دۆخى مرۆيى له ئۆكراینا، رۆڵى ديپلۆماسى و هه‌وڵه‌كانى چاره‌سه‌رى و كۆتاييهاتنى شه‌ڕ و پێكدادان، گرنگيى پاراستنى ئارامى و سه‌قامگيرى و گفتوگۆكانى ڤييه‌نا.  کۆبوونەوەی نێچیرڤان بارزانی و محەممەد بن عەبدولڕەحمان ئالسانی  له په‌راوێزى کۆڕبه‌نده‌که‌دا نێچيرڤان بارزانى بۆ گفتوگۆ لەباره‌ى دوايين پێشهاته‌كان و بابه‌تى هه‌نووكه‌يى، چه‌ند ديدار و كۆبوونه‌وه‌يه‌كی ئه‌نجامدا. لە یەکەم رۆژی گەیشتنی بۆ ئەنتاڵیا، نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان لەگەڵ محەممەد بن عەبدولڕەحمان ئالسانی، وەزیری دەرەوەی قەتەر و جێگری سەرۆکوەزیرانی ئەو وڵاتە کۆبووەوە. بەپێی راگەیێندراوێکی سەرۆکایەتیی هەرێمی کوردستان "دوايين پێشهاته‌كانى دۆخى ئۆكراينا و كاريگه‌رييه‌كانى شه‌ڕ و لێكه‌وته‌كانى، په‌يوه‌ندييه‌كانى عێراق و هه‌رێمى كوردستان له‌گه‌ڵ قه‌ته‌ر، دۆخى ئه‌فغانستان و ناوچه‌كه‌يان به گشتى تاوتوێ كرد".  فەلاح مستەفا، راوێژکاری باڵای سەرۆکی هەرێمی کوردستان بۆ کاروباری دەرەوە بە رووداوی راگەیاند، "کۆڕبەندەکە دەرفەتێکی گونجاو بوو بۆ سەرۆکی هەرێم بە هەموو ئەو کۆبوونەوانەی ئەنجامیدا، لەگەڵ سەرۆکوەزیران کە هاوکات وەزیری دەرەوەی قەتەریشە، کۆبووەوە. ئەوەش گرنگە بۆ هەرێمی کوردستان کە وڵاتێک لە کەنداو بە بایەخەوە دەڕوانێتە پەیوەندی لەگەڵ هەرێمی کوردستان و چۆنێتیی بەرەوپێشچوونی ئەو پەیوەندییە، ئەوانە پرسی گرنگ بوون بۆئەوەی باس بکرێن".  کۆبوونەوەی نێچیرڤان بارزانی و مەولوود چاوشئۆغڵو  بەپێی راگەیێندراوێکی سەرۆکایەتیی هەرێمی کوردستان، له كۆبوونه‌وه‌ی مەولوود چاوشئۆغڵو، وەزیری دەرەوەی تورکیا لەگەڵ نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان، دۆخى سياسيى عێراق، په‌يوه‌ندييه‌كانى هه‌ولێر و به‌غدا، بره‌ودان به په‌يوه‌ندييه‌كانى توركيا له‌گه‌ڵ عێراق و هه‌رێمى كوردستان، دوايين پێشهاته‌كانى شه‌ڕ و دۆخى ئۆكراينا و دۆخى ناوچه‌كه به‌گشتى، تاوتوێ كراون.  وەکو لە راگەیێندراوەکەی سەرۆکایەتیی هەرێمی کوردستاندا هاتووە، وه‌زيرى ده‌ره‌وه‌ى توركيا "سوپاسى به‌شداريى سه‌رۆك نێچيرڤان بارزانى له كۆڕبه‌ندى ديپلۆماسيى ئەنتاڵيادا كرد" و دووپاتیكرده‌وه كه توركيا به بايه‌خه‌وه له په‌يوه‌ندييه‌كانى له‌گه‌ڵ عێراق و هه‌رێمى كوردستان و له دۆخى سياسيى عێراق ده‌ڕوانێت.  لەبارەی کردنەوەی نووسینگەی نوێنەرایەتیی حکومەتی هەرێمی کوردستان لە ئەنقەرە گفتوگۆ کرا  لەبارەی کۆبوونەوەکەی نێچیرڤان بارزانی و مەولوود چاوشئۆغڵو؛ فەلاح مستەفا، راوێژکاری باڵای سەرۆکی هەرێمی کوردستان بۆ کاروباری دەرەوە بە رووداوی راگەیاند، " "سەرۆکی هەرێمی کوردستان سوپاسی دەستپێشخەریی تورکیای کرد بۆ نێوەندگیری و ئامادەباشی و زەمینەسازکردن بۆ دیالۆگی نێوان رووسیا و ئۆکراینا، چونکە دەرفەتێکە بۆ دیپلۆماسی و کۆتاییهێنان بە شەڕ".  هەروەها گوتیشی، گفتوگۆیان لەبارەی ئەوە کردووە کە چی بکرێت بۆ بەرەوپێشچوونی زیاتری پەیوەندییەکانی تورکیا و هەرێمی کوردستان، لەبارەی پرۆسەی سیاسیی عێراقیش گفتوگۆ کرا، بەرپرسانی تورکیا خوازیارن سەرکردە و لایەنە سیاسییەکانی عێراق بگەنە رێککەوتن و حکومەتێک پێکبهێنن کە وەڵامدانەوەی خواستی خەڵک بێت.  ئاماژەی بەوەش کرد، کۆڕبەندی دیپلۆماسیی ئەنتاڵیا دەرفەتێک بوو بۆئەوەی سەرۆکی هەرێمی کوردستان ئاگاداری پرسە جیهانییەکان بێت، وەکو پەتای کۆرۆنا، پرسە ئابوورییەکان، گۆڕانکاریی کەشوهەوا، تیرۆر، شەڕی ئۆکراینا و رووسیا، هەروەها دەرفەتێک بوو بۆ هەرێمی کوردستانیش لەگەڵ ئەو سەرکردانەی لە کۆڕبەندەکە ئامادەبوون بیروڕا و زانیاری بگۆڕنەوە.  لەبارەی کردنەوەی نوێنەرایەتیی حکومەتی هەرێمی کوردستان لە ئەنقەرە، ئەوەی خستەڕوو کە "ئاستی پەیوەندیی هەرێم [ی کوردستان] لەگەڵ تورکیا لە ئاستێکی باشدایە، ئەو پرسە پێشتریش باس کراوە و وەزارەتی دەرەوەش رەزامەندیی لەسەر داوە، ئەم کۆڕبەندەش دەرفەتێکی باش بوو کە دووبارە باسی لێوەبکرێت، وەزیری دەرەوەی تورکیاش لەوبارەوە گوتی، بەدواداچوون بۆ ئەو پرسە دەکەین".  هەروەها گوتیشی، "بابەتی گاز لەم دیدارەدا باس نەکرا، بەڵام پێشتر باس لەوە کراوە کە چۆن عێراق و هەرێمی کوردستان پێکەوە سوود لە سەرچاوە سرووشتییەکان وەربگرن بۆئەوەی لەڕێگەی تورکیاوە ببنە لایەنێکی دابینکەری ئەو سەرچاوە سرووشتییانە بۆ ئەو وڵاتانەی پێویستیان پێیەتی، ئەوەش دەرفەتێکی گرنگە بۆ کردنەوەی دەرگای هاوکاری بەڕووی یەکدیدا".  ئاماژەی بەوەش کرد، کۆڕبەندەکە، هەنگاوێکە بۆ نزیککردنەوەی ئەو وڵاتانەی وەکو یەک بیردەکەنەوە و هاوشێوەی یەکن، ئەوانەی بڕوایان بە دیپلۆماسی و کاری هاوبەشە بۆئەوەی نزیکببنەوە و کاری هاوبەشیان پێکەوە هەبێت.  کۆبوونەوەی نێچیرڤان بارزانی و رەجەب تەیب ئەردۆغان  نێچیرڤان بارزانی و رەجەب تەیب ئەردۆغان لە چوارچێوەی کۆڕبەندی دیپلۆماسیی ئەنتاڵیا کۆبوونەوە، لە کۆبوونەوەکەدا گفتوگۆ لەبارەی ده‌رفه‌ته‌كان و چۆنێتیی بره‌ودان به په‌يوه‌ندييه‌كانى توركيا له‌گه‌ڵ عێراق و هه‌رێمى كوردستان و فراوانكردنى بواره‌كانى هاريكاريى هاوبه‌ش و پێشخستنى زياترى جووڵه‌ى بازرگانى و ئابوورى كرد.  ئەردۆغان: دەرفەتێکی باشە دوایین پەرەسەندنەکان لە زاری سەرۆکی هەرێمی کوردستانەوە ببیستین  لەبارەی کۆبوونەوەکەی نێچیرڤان بارزانی و رەجەب تەیب ئەردۆغان، راوێژکاری باڵای سەرۆکی هەرێمی کوردستان بۆ کاروباری دەرەوە گوتی، "گفتوگۆیان لەبارەی دوایین پەرەسەندنەکانی عێراق و پرۆسەی سیاسی و پێکهێنانی حکومەتی نوێ و هەڵبژاردنی پۆستی سەرۆککۆماری ئەو وڵاتە کرد، سەرۆککۆماری تورکیا گوتی، دەرفەتێکی باشە دوایین پەرەسەندنەکان لە زاری سەرۆکی هەرێمی کوردستانەوە ببیستین".  هەروەها فەلاح مستەفا گوتیشی، تاوتوێی پەیوەندیی تورکیا لەگەڵ هەرێمی کوردستان و عێراق کرا، لەڕووی ئابووری و بازرگانی و دوایین پەرەسەندنەکانی پەیوەندییەکانیان، چونکە وەکو چۆن پرسی وڵاتانی دیکە کاریگەریی هەیە لەسەر عێراق و هەرێمی کوردستان، پرسی ئەوانیش کاریگەریی لەسەر وڵاتان هەیە.  ئەو راوێژکارەی سەرۆکی هەرێمی کوردستان ئاماژەی بەوەش کرد، "کۆڕبەندەکە دەرفەتێکی گونجاو بوو بۆ سەرۆکی هەرێمی کوردستان بە هەموو ئەو کۆبوونەوانەی ئەنجامیدا، لەگەڵ جێگری سەرۆکوەزیرانی قەتەر کە هاوکات وەزیری دەرەوەی قەتەریشە، کۆبووەوە، ئەوەش گرنگە بۆ هەرێمی کوردستان کە وڵاتێک لە کەنداو بە بایەخەوە دەڕوانێتە پەیوەندی لەگەڵ هەرێمی کوردستان و چۆنێتیی بەرەوپێشچوونی ئەو پەیوەندییە، ئەوانە پرسی گرنگ بوون بۆئەوەی باس بکرێن".  بەپێی راگەیێندراوی سەرۆکایەتیی هەرێمی کوردستان، لە کۆبوونەوەکەدا دۆخى عێراق و ئاڵنگارییەکانی به‌رده‌م پرۆسه‌ى سياسى له عێراق ته‌وه‌رێكى ديكه‌ى كۆبوونەوەكه بوو.  هه‌ردوولا هاوڕا بوون له‌سه‌ر گرنگيى گه‌يشتنى هێز و لايه‌نه عێراقييه‌كان به لێكگه‌يشتنێكى هاوبه‌ش و ده‌رچوون له چه‌قبه‌ستوويى سياسى و پێكهێنانى حكومه‌تى داهاتووى فيدراڵيى عێراق كه ره‌نگدانه‌وه‌ى پێكهاته‌كانى بێت.     هەروەها سەرۆکایەتیی تورکیا لە تویتەر وێنەی رەجەب تەیب ئەردۆغان و سەرۆک و سەرکردەی وڵاتان، لە نێویشیاندا نێچیرڤان بارزانی بڵاوکردووەتەوە و ئاماژەی بەوە داوە کە سەرۆککۆماری تورکیا لەگەڵیان کۆبووەتەوە.  کۆڕبەندەکە لەژێر ناونیشانی "داڕشتنەوەی دیپلۆماسی" بەڕێوەدەچێت و سێ رۆژ دەخایەنێت، لە رۆژانی ١١ تاوەکو ١٣ ـی ئەم مانگە بەردەوام دەبێت. ئەمڕۆ نێچیرڤان بارزانی کۆتایی بە بەشداریکردن و کۆبوونەوەکانی هێنا و گەڕایەوە بۆ هەولێر.  ئەمساڵ ١٥ سەرۆک و سەرۆکوەزیران و ٦٧ وەزیر بەشداریی کۆڕبەندەکە دەکەن و مژارەکانی دیدارەکە بریتین لە: هەژاری، ئاوارەبوون، گۆڕانی کەشوهەوا و هەڕەشەی گرووپە چەکدارەکان.
