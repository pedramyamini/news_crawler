مستەفا هیجری، لێپرسراوی گشتیی حیزبی دیموکراتی کوردستانی ئێران - حدکا لە تویتێکدا لەبارەی قسەکانی نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان رایگەیاند: "وتەکانی بەڕێز نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان لە کۆڕبەندی ئاسایشی رۆژهەڵاتی نێوەڕاست لە زانکۆی ئەمریکیی کوردستان لە دهۆک، بە گشتی و بە تایبەتی ئەو بەشەی کە پەیوەندیی بە چارەسەر کردنی پرسی کوردەوە هەبوو، لەسەر بنەمای واقعییەتێکە کە مێژوویەکی دوورودرێژی لە پشتە و جێبەجێ کردنەکەی بەشێکی گرنگە لە دابینبوونی ئاسایش و ئاشتی لەو وڵاتانەدا کە کوردستانیان بە سەردا دابەش کراوە".  لێپرسراوی گشتیی حیزبی دیموکراتی کوردستانی ئێران - حدکا هەروەها رایدەگەیێنێت: "ئەم بۆچوونە کە سەرنجی خرایە سەر، بەشێکە لە سیاسەت و هەڵوێستی هەمیشەیی و فەرمی ئێمە سەبارەت بە چارەسەرکردنی ئاشتیانەی پرسی کورد، هەر بۆیە جێی دەستخۆشییە".  دەقی قسەکانی نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان لەبارەی پرسی کورد لە زانکۆی ئەمریکیی کوردستان - دهۆک  ئەگەر تەماشای وڵاتانی ئێران، عێراق، سووریا و تورکیا سەبارەت بە پرسى کورد بکەین، دووبارە ئەمە پرسێکى ناوخۆییی هەریەک لەو وڵاتانەیە. پێویستە هەریەک لەو وڵاتانە لە چوارچێوەی سنوورى خۆیدا چارەسەركردنی كێشەكە بە گرنگییەوە وەرگرێت.  کورد لەو وڵاتانەدا، نابێ جیا بکرێتەوە، بەڵکو پێویستە جێیان بکرێتەوە. بەپێی سەنگی خۆی، هاوبەشىی و بەشداریی پێبکرێت و بواری پێبدرێت لە دروستکردنی بڕیاردا، هاوبەش و بەشدار بێت. نەتەوەکانی کورد و فارس، عەرەب و تورک، پێش ئەم دەوڵەتانەی ئێستا (لە تەمەنیاندا كە یەك سەدە دەگوزەرێت) هەر هەبوون، پێکەوە مێژوو و جوگرافیاى هاوبەشیان هەیە، بەدڵنیاییشەوە ئەم نەتەوانە وەک دراوسێ پێکەوە دەمێننەوە. بۆیە جگە لەیەکترقبووڵکردن و پێکەوەهەڵکردن، بژاردەیەکی دیکەمان نییە.  جەنگ و کوشتار و زمانی هێز، هەرچەند درێژە بکێشێت، لە کۆتاییدا هه‌ر دەبێ دەست بە دانوستان و گفتوگۆ بکرێتەوە. هەوڵی یەکترسڕینەوە، هەرچەند توند بێت و درێژە بکێشێت، لە کۆتاییدا هەر دەبێ یەکترى بخوێنینەوە و ئەو ڕاستییە بهێنینەوە بیری خۆمان کە کەسمان بڕیارمان نەداوە لێرە بڕۆین، کەسمان ناتوێینەوە. بۆیە، پێکەوەژیان لە چوارچێوەى یەک دەوڵەت لە یەک ناوچەدا، لەسەر بنەمای هاوبەش، تەنها ڕێگەی چارەسەرمانە.  ئێستاش کە نەتەوەکانی فارس و تورک و عەرەب لەم ناوچەیە خاوەن دەوڵەتن، ئەرکی ئەوانە مافی کورد و پێکهاتەکانى دیکە لە چوارچێوەی دەوڵەتەکانی خۆیاندا بدەن. مافی کولتووری، مافی سیاسی و مافی کۆمەڵایەتی.  بەرەی یەکگرتووی کورد: هاوڕای قسەکانی نێچیرڤان بارزانیین  رۆژی یەکشەممە ٢١ - ١١ - ٢٠٢١ بەرەی یەکگرتووی کورد لە کاردانەوەیدا بۆ گوتارەکەی نێچیرڤان بارزانی، سەرۆکی هەرێمی کوردستان لەبارەی پێدان ماف بە کورد لەلایەن دەسەڵاتەکانی فارس، تورک و عەرەبەوە، پەیامێکی بڵاوکردەوە و پێداگری لەسەر ئەو شێوازە چارەسەرییەی پرسی کورد دەکات، کە نێچیرڤان بارزانی باسی کردووە.  بەرەی یەکگرتووی کورد، کە تەنیا رێکخراوی سیاسیی نێوخۆی رۆژهەڵاتی کوردستانە، لە پەیامێکدا پشتیوانیی لە گوتاری نێچیرڤان بارزانی لە کۆڕبەندی ئاشتی و سەقامگیری رۆژهەڵاتی نێوڕاست لە دهۆک دەکات، کە پێداگریی لەسەر چارەسەریی کێشەی کورد لەلایەن حکومەتەکانی ئێران، تورکیا و سووریا کرد.  لە پەیامەکەی بەرەی یەکگرتووی کورددا ئاماژە بەوە کراوە، کە لە رۆژهەڵاتی نێوەڕاست، چوار نەتەوەی تورک، فارس، عەڕەب و کورد، زۆرترین ژمارەیان هەیە، بەڵام تەنیا نەتەوەی کوردە، کە بەهۆی دابەشبوونی بەسەر چوار وڵاتدا نەیتوانیوە ببێتە خاوەنی دەوڵەتی خۆی.   بەرەی یەکگرتووی کورد باس لەوە دەکات، کە گوتارەکەی نێچیرڤان بارزانی نیشانەی ئەزموونی سیاسی و واقعبینی ئەوە و دەڵێت: "گوتارەکەی چەند رۆژ لەمەوبەری نێچیرڤان بارزانی، کە داوای چارەسەری کێشەی کورد لەلایەن حکومەتەکانەوە دەکات، نیشانەی واقعبینی و بەرهەمی چەندین دەیە ئەزموونی سیاسییە، بەرەی یەکگرتووی کورد وەکوو بزووتنەوەیەکی سیاسی کە هەموو هەوڵەکانی بۆ جێگیرکردنی دادپەروەری و گەشەپێدان و دیموکراسییە، سەرەڕای پشتیوانیی لە گوتارەکەی بەڕێز نێچیرڤان بارزانی، تەنیا رێکاری گونجاو بۆ چارەسەری پرسی کورد، نەک لە رێگەی شەڕ و بەرەنگاربوونەوە و سەرکوتکردنی دەنگی بەرامبەر، بەڵکو لە رێگەی دایەلۆگ و دانوستاندن و ئاشتەوایی دەزانێت، کە ئەنجامەکەی دەبێتە هۆی بە فەرمی ناسینی مافی کورد و رەخساندنی بوارێکی گونجاو، کە کوردیش بۆ ئاوەدانی و گەشەپێدان و بەرزڕاگرتنی ئەم موڵک و گەلە هەوڵ بدات".
