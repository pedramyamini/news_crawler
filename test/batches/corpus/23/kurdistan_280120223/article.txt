لە هەرێمی کوردستان خانەی بەساڵاچووان تایبەتکراون بۆ ئەو کەسانەی کە هیچ جێگە و پەنایەکیان نییە، بەڵام رێنماییەکان ئەو کەسانە ناگرێتەوە کە تووشی نەخۆشیی دەرونی بوون یان کەسێک خۆی نەکات بەخاوەنیان. ئەمەش وایکردووە ئەو کەسانە لەو پەناگەیەش سوودمەند نەبن. زۆر جار کەسانی خۆبەخش و رێکخراوە مرۆییەکان بەهانای ئەو کەسانەوە دەچن کە دەسەڵات و توانای ئەوانیش سنووردارە.   کاژێر یەکی بەرەبەیانە، کەشوهەوا سێ پلە ژێر سفری سیلیزییە، ئەو بازاڕەی لە رۆژدا جمەی دێت، ئێستا چۆڵ و کپە، جگە لە پیاوێک کە لە پەنای دووکانێكی بازاڕەکەدا خەوتووە کەسی دیکە نابیندرێت. حامی ٢٢ ساڵە بەبێ جێگە و لانە لە شەقامەکانی سلێمانی ژیان دەکات. هیچی نییە جگە لە راخەرێک و بەتانییەکی کۆن، کەمترین کات قسە دەکات.  هاونیشتمانییەکی دیکە زیاتر لە ١٥ ساڵە، لەشەقام و بازاڕەکانی سلێمانی بەبێ جێگە و لانە دەسووڕێتەوە. هاوبەش یەکێکی دیکە لەوانەیە کە هەر لە منداڵییەوە بێ کەس دەبێت و هەر زوو دەکەوێتە سەر شەقام و بەهۆیەوە تووشی نەخۆشی دەرونی بووە.  هاوبەش موحسین دەڵێت: "زۆر سەرمام دەبێت کە دەخەوم لە سەرمادا خۆم لوول دەکەم و ناچار دەبم بەو سەرما بخەوم. زۆرجار وەکو مردووم لێدێت، دۆخم زۆر خراپە هێچ نابینم چاوم قورس دەبێت ناتوانم بیکەمەوە".  ئەوانەی بێلانەنن و لە شەقامەکانن دیار نیە کە چەندن چونکە هەرجارە و لە شارێکەوە دەچنە شارێکی دیکە، ئەوانەی ژیانی بێلانەکان دەبینن تێدەگەن چ مەینەتییەک دووچاریان بووەتەوە.  جەبار محەممەد، کرێکاری ناو بازاڕی سلێمانی دەڵێت: "بڕوابکەن دۆخیان زۆر خراپە رۆژی وا هەبووە هاتوون چوینەتە لایان جووڵەیان نەبووە وتومانە ئەمە مردووە، چونکە زۆر زۆر ساردبووە".  رەنگە ژیان بۆ ژنانی بێلانە سەختتر بێت لە پیاوان کە وەکو شایەتحاڵان دەڵێن، بەشێک لە ژنان لەلایەن پیاوانەوە خراپ
