له پلاتفۆرمی کوردستان بۆ پرسە نیشتمانییەکان سه‌باره‌ت به تاوتوێكردنی پرسی هه‌ڵبژاردنه‌كان، كه ئه‌مڕۆ شه‌ممه ٢٩ - ١ - ٢٠٢٢، له شارى سلێمانی به‌ئاماده‌بوونی سه‌رۆك فراكسیۆنه‌كانی په‌رله‌مان و ژماره‌یه‌ك پەرلەمانتار و شاره‌زایانی ئه‌كادیمی به‌ڕێوه‌چوو، هێمن هه‌ورامی، جێگرى سه‌رۆكی په‌رله‌مانی كوردستان، وتارێكی پێشكێشكرد و رایگەیاند، "ئێمه له په‌رله‌مانی كوردستان به‌رده‌وام پشتیوانیی ئه‌وه‌ین كه رێكخراوه‌كانی كۆمه‌ڵگەی مه‌ده‌نی، دامه‌زراوه ئه‌كادیمییه‌كان، به‌شداربن له‌و گفتوگۆیانه‌ی كه له‌سه‌ر پرسه نیشتیمانییه‌كان ده‌كرێن، بۆیه به‌شداربوونی ئێمه و به‌شێك له سه‌رۆك فراكسیۆنه‌كان و ئه‌ندامانی په‌رله‌مان له‌و چالاكییه، بۆ ده‌رخستنی پاڵپشتی و پشتیوانییه بۆ ئه‌و گفتوگۆیانه‌ی كه ئێستا له‌نێوان لایه‌نه سیاسییه‌كان و فراكسیۆنه‌كان هه‌یه بۆ رای گشتی و بۆ رێكخراوه‌كانی كۆمه‌ڵگەی مه‌ده‌نی و بۆ خه‌ڵكانی پسپۆڕ، زیاتر روونبكرێته‌وه به‌مه‌به‌ستی ده‌وڵه‌مه‌ندكردن و سوودوه‌رگرتن له بیروبۆچوونه جیاوازه‌كان و زه‌مینه‌سازكردنی لێكتێگه‌یشتنێكی هاوبه‌ش له‌م رووه‌وه".  هه‌ر له‌وتاره‌كه‌یدا د. هێمن هه‌ورامی، جێگرى سه‌رۆكی په‌رله‌مان، باسی له هه‌وڵه‌كانی په‌رله‌مانی كوردستان بۆ نووسینه‌وه‌ى ده‌ستوورى هه‌رێمی كوردستان كرد و جه‌ختى له‌سه‌ر پێویستیی پابه‌ندبوون به‌و كاتبه‌ندییانه كرده‌وه بۆ ئه‌نجامدانی هه‌ڵبژاردنه‌كان له كاتی خۆیدا و رایگه‌یاند: "ئێمه له سه‌رۆكایه‌تیی په‌رله‌مانی كوردستان هه‌ر له سه‌ره‌تای ده‌ستبه‌كاربوونمان له مانگی ٨ ى ٢٠١٩ ـه‌وه سه‌ردانی ته‌واوی لایه‌نه سیاسییه‌كانى كوردستانمان كردووه بۆ ئه‌وه‌ی له‌سه‌ر بابه‌تی ده‌ستوور، له‌سه‌ر بابه‌تی لێكتێگه‌یشتن، له‌سه‌ر بابه‌تی خۆئاماده‌كردن، تاوەکو بتوانین زه‌مینه‌سازییه‌كی باشترى سیاسی و یاسایی بكرێت، بۆ ئه‌وه‌ی ده‌ستوورى هه‌رێمی كوردستان له‌م خوله‌ى په‌رله‌مانی كوردستان كارى له‌سه‌ر بكرێت، هه‌ر له‌چوارچێوه‌ی ئه‌مه‌شدا ئێمه له سه‌رۆكایه‌تیی په‌رله‌مانی كوردستان هه‌میشه له كۆبوونه‌وه‌ى سه‌رۆك فراكسیۆنه‌كان، له كۆبوونه‌وه و دانیشتنه‌كان، له ده‌ستپێكی وه‌رزه‌كان، له كۆبوونه‌وه‌ى سێ سه‌رۆكایه‌تییه‌كه‌ى هه‌رێمی كوردستان، رژد بووین له‌سه‌ر ئه‌وه‌ى كه لایه‌نه سیاسییه به‌ڕێزه‌كان، فراكسیۆنه‌كانى په‌رله‌مانی كوردستان، پابه‌ندی ئه‌و كاتبه‌ندییانه بن كه پرۆسه‌ی دیموكراسی، هه‌ڵبژاردن و بابه‌ته‌كانی دیکە كه پێویست ده‌كات ئاماده‌كاریى زیاتریان بۆ بكرێت، به‌ڵام له‌گه‌ڵ نزیكبوونه‌وه‌ى واده‌كان، ئێمه رۆژی سێشه‌ممه ٣ - ١١ - ٢٠٢١، وه‌كو سه‌رۆكایه‌تیی په‌رله‌مانی كوردستان كۆبوونه‌وه‌مان له‌گه‌ڵ سه‌رجه‌م فراكسیۆنه به‌ڕێزه‌كان هه‌بووه و له‌وێ به وردى و به روونی پێمان گوتوون فراكسیۆنه‌كان، لایه‌نه سیاسییه‌كان، له‌گه‌ڵ سه‌ركردایه‌تییه‌كانی خۆتان قسه بكه‌ن كه ئێمه وادەیەکی ده‌ستووریمان هه‌یه ٦ - ١١ - ٢٠٢٢، واده‌ی خولی پێنجه‌می په‌رله‌مانى كوردستان كۆتایی پێدێت، واته ده‌بێت پێش ٥ - ١١ - ٢٠٢٢، هه‌ڵبژاردن بۆ خولی شه‌شه‌می په‌رله‌مانی كوردستان كرابێت، بۆ ئه‌مه‌ش به‌پێی یاسای سه‌رۆكایه‌تیی هه‌رێمی كوردستان له ٦ - ٩ - ٢٠٢٢، سه‌رۆكی هه‌رێمی کوردستان دەبێ فه‌رمانی هه‌رێمیی بۆ دیاریكردنی رۆژی هه‌ڵبژاردن ده‌ركردبێت و دەبێ كۆمیسیۆنی سه‌ربه‌خۆی هه‌ڵبژاردنه‌كانیش پێش ٩/٦ به شه‌ش مانگ، واته له ٣/٦ كۆمیسیۆن ئاماده بووبێت بۆئه‌وه‌ی هه‌ڵبژاردنه‌كان له كاتی خۆیدا بكرێت، له‌م رووه‌شه‌وه ئێمه جارێكی دیکەش رۆژی پێنجشه‌ممه ٢٠٢١/١١/١٨، كۆبوونه‌وه‌یه‌كی دیکەی سه‌رۆكایه‌تیی په‌رله‌مان و سه‌رۆك فراكسیۆنه‌كانمان كردووه بۆ له‌یه‌كتر نزیككردنه‌وه‌ى بیروبۆچوونی ئه‌و لایه‌نانه تاوەکو بتوانین خاڵی هاوبه‌ش له بیروبۆچوونه‌كانیان بدۆزینه‌وه بۆ ئه‌وه‌ی میكانیزمێك بگیرێته‌به‌ر كه هه‌موو لایه‌ك به‌شداربن له ئاڕاسته‌كردن و پابه‌ندبوون به‌م وادە ده‌ستوورییانه‌ی كه هه‌یه".  جێگری سەرۆکی پەرلەمانی کوردستان ئاماژەی بەوەشکرد "جارێكی دیکە ئێمه له سه‌رۆكایه‌تیی په‌رله‌مانی كوردستان، رۆژی پێنجشه‌ممه ٢٠٢١/١٢/١٦، كۆبوونه‌وه‌یه‌كمان له‌گه‌ڵ سه‌رجه‌م سه‌رۆك فراكسیۆنه‌كان كردووه بۆئه‌وه‌ى ده‌رفه‌تی زیاتر بۆ گفتوگۆكردن، بۆ لێكتێگه‌یشتن، بۆ گه‌یشتنه میكانیزمێكی گونجاو تاوەکو بتوانین هه‌موومان به‌یه‌كه‌وه پابه‌ند و ئه‌و به‌رپرسیاریه‌تییه یاساییه‌ی له‌سه‌ر شانمانه جێبه‌جێی بكه‌ین، بۆ ئه‌وه‌ی فراكسیۆنه‌كان زیاتر ده‌رفه‌تی گفتوگۆكردنیان هه‌بێت، داوامان لێكردن كه به‌بێ ئاماده‌بوونی ده‌سته‌ی سه‌رۆكایه‌تی جارێكی دیکە له‌نێوان خۆیاندا كۆببنه‌وه، رۆژی ٢٠٢١/١٢/٢٠، سه‌رۆك فراكسیۆنه‌كان به‌بێ ئاماده‌بوونی ده‌سته‌ی سه‌رۆكایه‌تی كۆبوونه‌وه‌یه‌كی خۆیانیان كردووه و له كۆبوونه‌وه‌كه‌یاندا گفتوگۆی جدییان له‌سه‌ر ئه‌و میكانیزمانه كردووه".  له به‌شێكی دیكه‌ی گوتاره‌كه‌یدا هێمن هەورامی، باسی ئامانجه‌كانی ده‌سته‌ی سه‌رۆكایه‌تیی په‌رله‌مانی کوردستان له‌م كۆبوونه‌وانه و سه‌رپه‌رشتیكردنیان كرد و رایگه‌یاند: "ئامانجی ئێمه له ده‌سته‌ی سه‌رۆكایه‌تیی په‌رله‌مانی كوردستان، له داڕشتنی هه‌موو ئه‌و كۆبوونه‌وانه و سه‌رپه‌رشتیكردن و زه‌مینه‌سازی، بۆ ئه‌وه بووه كه دووپاتی بكه‌ینه‌وه هه‌موو لایه‌كمان پابه‌ند بین به‌و كاتبه‌ندییه ده‌ستوورییانه‌ی كه هه‌یه، بۆئه‌وه‌ی هه‌موومان هاوكاربین له ئه‌نجامدانی هه‌ڵبژاردنه‌كان له كاتی خۆی و ئه‌و ده‌رفه‌ته سیاسی و یاساییانه‌ی كه پێویست ده‌كا، ئه‌و ئاماده‌كارییه ته‌كنیكییانه‌ی كه پێویست ده‌كا، چ له نوێكردنه‌وه‌ى كۆمیسیۆن بێت یان هه‌مواركردنه‌وه‌ى یاسای كۆمیسیۆن و پێكهێنانی كۆمیسیۆنێكی نوێ بێت، یان ده‌نگدانه‌وه و متمانه‌دانه‌وه به‌و كۆمیسیۆنه‌ى ئێستا و پڕكردنه‌وه‌ی بۆشاییەکانی بێت، هه‌ر بابه‌تێك كه فراكسیۆنه‌كان له‌سه‌ری كۆكبن، ئێمه كارى له‌سه‌ر بكه‌ین".  هێمن هەورامی دەڵێت "ئه‌مڕۆ ٢٠٢٢/١/٢٩، واته كاتی پێویست درایه سه‌رۆك فراكسیۆنه‌كان، كاتی پێویست دراوه‌ته لایه‌نه سیاسییه‌كان، ئێمه بۆ ئه‌وه‌ی بتوانین پابه‌ندى ئه‌و وادە ده‌ستوورییانه بین و پێش ٢٠٢٢/١١/٦ هه‌ڵبژاردن بۆ خولی شه‌شه‌می په‌رله‌مانی كوردستان كرابێت، كاتێكی زۆرمان به‌ده‌سته‌وه نه‌ماوه، بۆیه ئێستا له ئه‌مڕۆ به‌دواوه كاتی كاركردن و بڕیاردانه له‌لایه‌ن فراكسیۆنه‌كانه‌وه بۆ ئه‌وه‌ى بتوانین هه‌موومان پابه‌ندبین به ته وادەی ده‌ستوورییه‌وه".  د. هێمن هەورامی ئاماژەی بەوەکرد، "ئه‌وه‌ى بۆ ئێمه گرنگه، ئه‌وه‌ى ئێستا كه بۆشاییمان هه‌یه، بۆشاییمان له كۆمیسیۆن هه‌یه، چونكه ئه‌و كۆمیسیۆنه‌ی كه ئێستا هه‌یه له ٢٠١٩/١٢/٢٠، كاته‌كه‌ی به‌سه‌رچووه، له‌به‌رئه‌وه‌ى له ٢٠١٤/١٢/٢٠ متمانه‌ى پێدراوه، ئه‌و كات كه كۆمیسیۆن دروستكراوه، ئه‌م كۆمسیۆنه به رێككه‌وتنی پێنج حزبی سیاسی بووه كه له‌ناو كابینه‌ی هه‌شته‌می حكومه‌تی هه‌رێمى كوردستاندا بوون، ئه‌وانیش پارتی دیموكراتی كوردستان، یه‌كێتیی نیشتیمانی، بزووتنه‌وه‌ى گۆڕان، یه‌كگرتووی ئیسلامی و كۆمه‌ڵی دادگەریی بوون، به‌پێی ئه‌و رێككه‌وتنه‌ى كه كردوویانه و به‌پێی به‌ركه‌وته‌ى هه‌ڵبژاردن و سانت لیگۆ، (٩) كۆمیسیاره‌كه و ته‌نانه‌ت ته‌واوى فه‌رمانبه‌ران و كارگوزارانی ئه‌م كۆمیسیۆنه به‌پێی به‌ركه‌وته‌ى هه‌ڵبژاردن له‌نێوان ئه‌و پێنج لایه‌نه سیاسییه‌دا به‌شكراون، پرسیاره‌كه ئه‌مه‌یه و به‌شوێن وه‌ڵامدا ده‌گه‌ڕێین، ئایا ئه‌م كۆمیسیۆنه متمانه‌ى پێبدرێته‌وه یان كۆمیسیۆنى نوێ دروست ده‌كرێته‌وه، یان یاسای كۆمیسیۆن ئه‌گۆڕدرێ و هه‌موار ده‌كرێت، بۆئه‌وه‌ی بیر له پێكهاته‌ى دیکەی كۆمیسیۆن بكرێته‌وه".  سه‌باره‌ت به دیاریكردنی چاره‌نووسی كۆمیسیۆنی باڵای سه‌ربه‌خۆی هه‌ڵبژاردنه‌كانیش، هێمن هه‌ورامی، رایگه‌یاند: "كۆمیسیۆنی سه‌ربه‌خۆی هه‌ڵبژاردنه‌كان، به نووسراوی ژماره ٧٦ له ٢٠١٩/٨/٢٥ داوایان له سه‌رۆكایه‌تیی په‌رله‌مانى كوردستان كردووه كه چاره‌نووسی كۆمیسیۆن به نه‌زه‌رى ئیعتیبار وه‌ربگرن و كارى له‌سه‌ر بكه‌ن، جارێكی دیکە به نووسراوی ژماره ٦٢ له ٢٠٢١/٨/٤ كۆمیسیۆنی سه‌ربه‌خۆی هه‌ڵبژاردنه‌كان داوایان له په‌رله‌مانی كوردستان كردووه كه چاره‌نووسیان یه‌كلایی بكاته‌وه بۆئه‌وه‌ى ئاماده‌كارییه‌كان بۆ ئه‌نجامدانی هه‌ڵبژاردن بكه‌ن".  جێگری سەرۆکی پەرلەمانی کوردستان گوتیشی: "ئێمه له سه‌رۆكایه‌تیی په‌رله‌مانی كوردستانه‌وه لێره‌وه رایده‌گه‌یێنین، پابه‌ندین و كاریشی بۆ ئه‌كه‌ین تاوەکو هه‌ڵبژاردنه‌كان له كاتی خۆیدا بكرێت و ئه‌و كاته‌ی كه پێویست بووه دراوه‌ته هه‌موو لایه‌نه سیاسییه‌كان و فراكسیۆنه‌كان، ئێمه له سه‌رۆكایه‌تی رێگه ناده‌ین په‌نجه‌ی تۆمه‌تمان بۆ درێژ بكرێت و بڵێن سه‌رۆكایه‌تیی په‌رله‌مان جیددى نه‌بوو له به‌ڕێوه‌بردن و زه‌مینه‌سازیكردن، بۆیه داوا له فراكسیۆنه‌كان و لایه‌نه سیاسییه‌كان ده‌كه‌ین، جیددییه‌تی كه‌میی ماوه و كه‌میی كات له‌به‌رچاو بگرن بۆ ئه‌وه‌ی هه‌موومان پێكه‌وه ئاماده‌سازیی جیددیترى بۆ بكه‌ین".  له كۆتایی گوتاره‌كه‌شیدا جێگرى سه‌رۆكی په‌رله‌مان رایگه‌یاند: "پێش ماوه‌یه‌كیش كۆبوونه‌وه‌یه‌كی گرنگی نێوان سه‌رۆكی هه‌رێمی كوردستان و جینین پلاسخارت، نێرده‌ی تایبه‌تیی سكرتێری گشتیی نه‌ته‌وه یه‌كگرتووه‌كان ئه‌نجامدرا و له‌و كۆبوونه‌وه‌یه‌شدا جه‌خت له‌وه كراوه‌ته‌وه كه ئه‌بێ هه‌ڵبژاردنه‌كان له كاتی خۆیدا بكرێت، بۆیه ئه‌و رێوشوێنه یاساییانه‌ی كه به‌پێی یاسای سه‌رۆكایه‌تیی هه‌رێمی کوردستان پێویستە ده‌گیرێنه به‌ر و ئێمه هه‌موو هه‌وڵێكیشمان دەده‌ین هه‌م له رێی ئێوه، له رێی ئه‌كادیمییه‌كان، لایه‌نه سیاسییه‌كان، فراكسیۆنه‌كان، یۆنامی، كۆمیسیۆنى عێراق، پێكه‌وه كار بكه‌ین له‌سه‌ر ئه‌وه‌ى كه بتوانین ئه‌وپه‌ڕی شه‌فافییه‌ت له ئاماده‌كارى بۆ پرۆسه‌ی هه‌ڵبژاردنه‌كان هه‌بێ و چاودێرییه‌كی وردى چۆنیه‌تیی ئه‌نجامدانی پرۆسه‌ی هه‌ڵبژاردنه‌كان هه‌بێت تاوەکو به‌یه‌كه‌وه بتوانین پرۆسه‌ی سیاسیی هه‌رێمى كوردستان به‌دوانه‌خستنی هه‌ڵبژاردنه‌كان، به ئه‌نجامدانی له كاتی خۆی ده‌رفه‌ت بدرێته هاووڵاتییانى كوردستان بۆ له بێژینگدانی هێزه سیاسییه ده‌سه‌ڵاتداره‌كانى ئێستا و هێزه‌كانی ئۆپۆزسیۆن و ده‌رفه‌تیش بداته ده‌نگ و ڕه‌نگی جیاوازتر كه ئێستا له‌ناو پرۆسه‌ی سیاسیی هه‌رێمی كوردستاندا هه‌ن، تا بتوانن له هه‌ڵبژاردنی ئاینده ده‌نگی پێویست به‌ده‌ستبێنن و بێنه ناو پرۆسه‌كه‌وه".
